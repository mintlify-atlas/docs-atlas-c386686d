```tsx

import { useState } from "react"

  const [password, setPassword] = useState("")

  return (
    <PasswordStrengthMeter

      onValueChange={setPassword}

      showPasswordToggle
    />
```

### With Custom Requirements

Define custom password requirements:

```tsx
import { PasswordStrengthMeter } from "@/components/ui/strength-meter"
import type { PasswordStrengthRequirement } from "@/components/ui/strength-meter"
import { useState } from "react"

const customRequirements: PasswordStrengthRequirement[] = [
  {
    label: "At least 12 characters",
    validator: (password) => password.length >= 12,
  },
  {
    label: "Contains uppercase and lowercase",
    validator: (password) => /[a-z]/.test(password) && /[A-Z]/.test(password),
  },
  {
    label: "Contains a number",
    validator: (password) => /\d/.test(password),
  },
  {
    label: "No common patterns",
    validator: (password) => !/123456|password|qwerty/i.test(password),
  },
]

export function CustomRequirementsMeter() {
  const [password, setPassword] = useState("")

  return (
    <PasswordStrengthMeter
      value={password}
      onValueChange={setPassword}
      requirements={customRequirements}
      showRequirements
    />
  )
}
```

### Custom Strength Calculation

Implement custom password strength calculation:

```tsx
import { PasswordStrengthMeter } from "@/components/ui/strength-meter"
import { useState } from "react"

const calculateCustomStrength = (password: string): number => {
  let score = 0
  
  // Length scoring
  if (password.length >= 8) score += 20
  if (password.length >= 12) score += 20
  if (password.length >= 16) score += 10
  
  // Character variety
  if (/[a-z]/.test(password)) score += 15
  if (/[A-Z]/.test(password)) score += 15
  if (/\d/.test(password)) score += 10
  if (/[^a-zA-Z\d]/.test(password)) score += 10
  
  return Math.min(score, 100)
}

export function CustomStrengthMeter() {
  const [password, setPassword] = useState("")

  return (
    <PasswordStrengthMeter
      value={password}
      onValueChange={setPassword}
      customCalculateStrength={calculateCustomStrength}
    />
  )
}
```

## API Reference

### PasswordStrengthMeter

<ParamField path="value" type="string" default="''">
  Current password value.
</ParamField>

<ParamField path="onValueChange" type="(value: string) => void">
  Callback function called when the password changes.
</ParamField>

<ParamField path="showText" type="boolean" default="true">
  Display the strength level text (Weak, Fair, Good, Strong).
</ParamField>

<ParamField path="showRequirements" type="boolean" default="true">
  Display the list of password requirements with check/cross icons.
</ParamField>

<ParamField path="segments" type="number" default="4">
  Number of segments in the strength meter bar.
</ParamField>

<ParamField path="strengthThresholds" type="Record<StrengthLevel, number>">
  Custom thresholds for each strength level. Default:
  ```tsx
  {
    empty: 0,
    weak: 1,
    fair: 40,
    good: 70,
    strong: 90,
  }
  ```
</ParamField>

<ParamField path="requirements" type="PasswordStrengthRequirement[]">
  Array of password requirements. Each requirement has a label and validator function.
  Default requirements check for: 8+ characters, lowercase, uppercase, number, special character.
</ParamField>

<ParamField path="customCalculateStrength" type="(password: string) => number">
  Custom function to calculate password strength (should return 0-100).
</ParamField>

<ParamField path="showPasswordToggle" type="boolean" default="true">
  Show eye icon to toggle password visibility.
</ParamField>

<ParamField path="strengthLabels" type="Record<StrengthLevel, string>">
  Custom labels for each strength level. Default: Empty, Weak, Fair, Good, Strong.
</ParamField>

<ParamField path="enableAutoGenerate" type="boolean" default="false">
  Enable automatic password generation button.
</ParamField>

<ParamField path="autoGenerateLength" type="number" default="10">
  Length of auto-generated passwords.
</ParamField>

<ParamField path="placeholder" type="string" default="'Enter password'">
  Placeholder text for the input field.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes for the container.
</ParamField>

<ParamField path="inputClassName" type="string">
  Additional CSS classes for the input field.
</ParamField>

<ParamField path="meterClassName" type="string">
  Additional CSS classes for the meter bar.
</ParamField>

<ParamField path="theme" type="StrengthMeterTheme">
  Custom theme object for styling. See StrengthMeterTheme interface.
</ParamField>

<ParamField path="label" type="string" required>

<ParamField path="validator" type="(password: string) => boolean" required>

## Features

- **Real-Time Validation**: Instant feedback as users type
- **Visual Strength Meter**: Segmented bar with color-coded strength levels
- **Requirement Checklist**: Clear indication of met/unmet requirements
- **Password Visibility Toggle**: Show/hide password with eye icon
- **Auto-Generate**: Built-in strong password generator
- **Custom Requirements**: Define your own validation rules
- **Custom Strength Calculation**: Implement custom scoring algorithms
- **Theming Support**: Fully customizable appearance
- **Accessible**: ARIA labels and keyboard navigation
- **TypeScript**: Full type safety and IntelliSense support

## Default Requirements

The component comes with these default password requirements:

1. At least 8 characters
2. At least one lowercase letter
3. At least one uppercase letter
4. At least one number
5. At least one special character (!@#$%^&*(),.?":{}|&lt;&gt;)

## Strength Calculation

The default strength calculation considers:

- **Base Score**: Percentage of requirements met
- **Length Bonus**: +10 for 12+ chars, +10 for 16+ chars
- **Complexity Bonus**: +10 for multiple special characters
- **Maximum**: Capped at 100

## Dependencies

This component requires the following dependencies:

- `input` - shadcn/ui input component
- `lucide-react` - For icons (Eye, EyeOff, Check, X, RefreshCw)

<Note>
  The auto-generated passwords include at least one character from each category (lowercase, uppercase, number, special) and are randomly shuffled for security.
</Note>

  Client-side password strength checking is for user guidance only. Always enforce password policies on the server side as well.
