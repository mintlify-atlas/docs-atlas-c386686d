## Your First Component

This guide will walk you through installing and using your first RigidUI component. We'll use the **Text Shift** component, which creates beautiful animated text transitions.

<Steps>
  <Step title="Install the Component">
    First, install the Text Shift component using the shadcn CLI:
    
    <CodeGroup>
    ```bash npm
    npx shadcn@latest add https://rigidui.com/r/text-shift.json
    ```
    
    ```bash yarn
    npx shadcn@latest add https://rigidui.com/r/text-shift.json
    ```
    
    ```bash pnpm
    npx shadcn@latest add https://rigidui.com/r/text-shift.json
    ```
    </CodeGroup>
    
    This will:
    - Add `text-shift.tsx` to your `components/ui` directory
    - Install the `motion` dependency (if not already installed)
    - Set up all necessary utilities
  </Step>
  
  <Step title="Import the Component">
    Import the Text Shift component in your page or component:
    
    ```tsx app/page.tsx
    import TextShift from "@/components/ui/text-shift"
    ```
  </Step>
  
  <Step title="Use the Component">
    Add the Text Shift component to your UI:
    
    ```tsx app/page.tsx
    export default function Home() {
      return (
        <main className="flex min-h-screen items-center justify-center">
          <TextShift
            primaryText="Hello World"
            secondaryText="Welcome to RigidUI"
            className="text-4xl"
          />
        </main>
      )
    }
    ```
  </Step>
  
  <Step title="Run Your App">
    Start your development server and see the animated text in action:
    
    <CodeGroup>
    ```bash npm
    npm run dev
    ```
    
    ```bash yarn
    yarn dev
    ```
    
    ```bash pnpm
    pnpm dev
    ```
    </CodeGroup>
    
    Hover over the text to see the smooth transition animation!
  </Step>
</Steps>

<Tip>
  The Text Shift component animates letter-by-letter with a staggered effect, creating a smooth and engaging transition between two text strings.
</Tip>

## Complete Example

Here's a complete working example with all the code you need:

```tsx app/page.tsx
"use client"

import TextShift from "@/components/ui/text-shift"

export default function Home() {
  return (
    <main className="flex min-h-screen flex-col items-center justify-center gap-8 p-24">
      <h1 className="text-2xl font-bold text-gray-800 dark:text-gray-200">
        Hover over the text below
      </h1>
      
      <TextShift
        primaryText="Hello World"
        secondaryText="Welcome to RigidUI"
        className="text-6xl"
        primaryTextColor="text-black dark:text-white"
        secondaryTextColor="text-purple-600 dark:text-purple-400"
        duration={0.8}
        staggerDelay={0.04}
      />
      
      <p className="text-gray-600 dark:text-gray-400 max-w-md text-center">
        This is the Text Shift component from RigidUI. It creates beautiful 
        animated transitions between two text strings with staggered letter animations.
      </p>
    </main>
  )
}
```

## Customizing the Component

The Text Shift component accepts several props for customization:

<ResponseField name="primaryText" type="string" required>
  The initial text to display
</ResponseField>

<ResponseField name="secondaryText" type="string" required>
  The text to transition to on hover
</ResponseField>

<ResponseField name="className" type="string">
  Additional CSS classes for styling
</ResponseField>

<ResponseField name="primaryTextColor" type="string" default="text-black">
  Tailwind class for primary text color
</ResponseField>

<ResponseField name="secondaryTextColor" type="string" default="text-gray-400">
  Tailwind class for secondary text color
</ResponseField>

<ResponseField name="duration" type="number" default={1}>
  Animation duration in seconds
</ResponseField>

<ResponseField name="staggerDelay" type="number" default={0.05}>
  Delay between each letter's animation in seconds
</ResponseField>

<ResponseField name="easing" type="Easing" default="[0.165, 0.84, 0.44, 1]">
  Motion easing function for the animation
</ResponseField>

    ```tsx app/page.tsx
          </DataReveal>
        </div>
      )
    }
    ```
  </Step>
</Steps>

<Note>
  The Data Reveal component uses a compound component pattern. This allows you to compose the button, container, and items flexibly while maintaining shared state.
</Note>

## Component Composition

The Data Reveal component demonstrates RigidUI's compound component pattern:

- `<DataReveal>` - Root component that manages state
- `<DataReveal.Button>` - Toggle button to reveal/hide
- `<DataReveal.Container>` - Container for revealed items with animations
- `<DataReveal.Item>` - Individual items that appear with staggered animations

```tsx Example with Props
<DataReveal 
  defaultRevealed={false}
  autoHide={true}
  autoHideDelay={3000}
  onToggle={(isRevealed) => console.log('Revealed:', isRevealed)}
>
  <DataReveal.Button className="bg-blue-500 text-white">
    Click to Reveal
  </DataReveal.Button>
  
  <DataReveal.Container className="gap-3">
    <DataReveal.Item content="Hidden" index={0} className="bg-gray-100" />
    <DataReveal.Item content="Content" index={1} className="bg-gray-100" />
    <DataReveal.Item content="Here" index={2} className="bg-gray-100" />
  </DataReveal.Container>
</DataReveal>
```

## Common Patterns

### Client-Side Components

Many RigidUI components use client-side features (animations, state, events). Make sure to use the `"use client"` directive:

```tsx app/components/my-component.tsx
"use client"

import TextShift from "@/components/ui/text-shift"

export default function MyComponent() {
  return (
    <TextShift 
      primaryText="Client"
      secondaryText="Component"
    />
  )
}
```

### Dark Mode Support

All RigidUI components support dark mode. Use Tailwind's dark mode classes:

```tsx
<TextShift
  primaryText="Light Mode"
  secondaryText="Dark Mode"
  primaryTextColor="text-gray-900 dark:text-white"
  secondaryTextColor="text-blue-600 dark:text-blue-400"
/>
```

### Combining Components

RigidUI components work great together:

```tsx
import { DataReveal } from "@/components/ui/data-reveal"
import TextShift from "@/components/ui/text-shift"

export default function Dashboard() {
  return (
    <div className="space-y-8">
      <TextShift
        primaryText="Dashboard"
        secondaryText="Welcome Back"
        className="text-5xl"
      />
      
      <DataReveal>
        <DataReveal.Button text="View Credentials" />
        <DataReveal.Container>
          <DataReveal.Item content="API Key" index={0} />
          <DataReveal.Item content="Secret" index={1} />
        </DataReveal.Container>
      </DataReveal>
    </div>
  )
}
```
